## 合并两个数组

```js
<script>
  // 合并两个数组
  var arr = [1, 2, 3]
  var brr = [4, 5, 6]

  // 方法1：
  Array.prototype.push.apply(arr, brr)
  console.log(arr);

  // 方法2：
  arr.push(brr)
  console.log(arr);

  // 方法3：
  arr = [...arr, ...brr]
  console.log(arr);
</script>
```



## 求数组中的最大值

```javascript
<!-- 求数组中的最大值 -->
<script>
  var nums = [45, 12, 46, 77, 35, 66, 55, 100]
  const maxNUM = Math.max.apply(Math, nums)
  console.log(maxNUM);
</script>
```
