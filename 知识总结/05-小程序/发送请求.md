# 发送请求

## GET

```javascript
tagget: function () {
  console.log("点击了")
  wx.request({
    url: "https://www.liulongbin.top:8081/api/get",   // 地址
    data: {},   // data 请求参数
    success: function () {}   // 响应结果
  })
},
```

`案例：`

```javascript
tagget: function () {
  console.log("点击了")
  wx.request({
    url: "https://www.liulongbin.top:8081/api/get",   // 地址
    // data 请求参数
    data: {
      username: "admin",
      password: "123456"
    },
    // 响应结果
    success: function (res) {
      console.log("成功的结果", res)
    }
  })
},
```

## POST

```javascript
tagpost: function () {
  console.log("点击了")
  wx.request({
    url: "https://www.liulongbin.top:8081/api/post",   // 地址
    method: "post",   // 请求类型
    // 请求参数
    data: {
      username: "admin",
      password: "123456"
    },
    // 响应结果
    success: function (res) {
      console.log("成功的结果", res)
    }
  })
},
```
